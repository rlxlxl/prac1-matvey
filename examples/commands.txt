# Примеры SQL команд для СУБД

## Инициализация
При запуске СУБД автоматически создается структура базы данных на основе schema.json

## SELECT - Выборка данных

### Простой SELECT из одной таблицы
SELECT таблица1.колонка1, таблица1.колонка2 FROM таблица1

### SELECT всех колонок из таблицы
SELECT таблица1.таблица1_pk, таблица1.колонка1, таблица1.колонка2, таблица1.колонка3, таблица1.колонка4 FROM таблица1

### SELECT с декартовым произведением (JOIN нескольких таблиц)
SELECT таблица1.колонка1, таблица2.колонка1 FROM таблица1, таблица2

### SELECT с WHERE условием (равенство)
SELECT таблица1.колонка1, таблица1.колонка2 FROM таблица1 WHERE таблица1.колонка1 = 'значение'

### SELECT с WHERE и AND
SELECT таблица1.колонка1, таблица2.колонка1 FROM таблица1, таблица2 WHERE таблица1.колонка1 = таблица2.колонка1 AND таблица1.колонка2 = 'string'

### SELECT с WHERE и OR
SELECT таблица1.колонка1, таблица1.колонка2 FROM таблица1 WHERE таблица1.колонка1 = 'value1' OR таблица1.колонка2 = 'value2'

### SELECT с несколькими условиями AND и OR
SELECT таблица1.колонка1, таблица1.колонка2 FROM таблица1 WHERE таблица1.колонка1 = 'test' AND таблица1.колонка2 = 'data' OR таблица1.колонка3 = 'other'

## INSERT - Вставка данных

### Вставка одной строки
INSERT INTO таблица1 VALUES ('значение1', 'значение2', 'значение3', 'значение4')

### Вставка строки с числовыми значениями
INSERT INTO таблица1 VALUES ('123', '456', '789', '012')

### Вставка в таблицу с меньшим количеством колонок
INSERT INTO таблица2 VALUES ('значение1', 'значение2')

### Вставка нескольких строк (выполнять по одной)
INSERT INTO таблица1 VALUES ('data1', 'data2', 'data3', 'data4')
INSERT INTO таблица1 VALUES ('data5', 'data6', 'data7', 'data8')

## DELETE - Удаление данных

### Удаление всех строк из таблицы (без WHERE)
DELETE FROM таблица1

### Удаление строк по условию
DELETE FROM таблица1 WHERE таблица1.колонка1 = '123'

### Удаление с условием AND
DELETE FROM таблица1 WHERE таблица1.колонка1 = 'value1' AND таблица1.колонка2 = 'value2'

### Удаление с условием OR
DELETE FROM таблица1 WHERE таблица1.колонка1 = 'value1' OR таблица1.колонка2 = 'value2'

## Примеры комплексных запросов

### 1. Создание и выборка данных
INSERT INTO таблица1 VALUES ('test1', 'data1', 'info1', 'desc1')
INSERT INTO таблица1 VALUES ('test2', 'data2', 'info2', 'desc2')
SELECT таблица1.колонка1, таблица1.колонка2 FROM таблица1

### 2. Соединение таблиц с фильтрацией
INSERT INTO таблица1 VALUES ('match', 'value1', 'info', 'desc')
INSERT INTO таблица2 VALUES ('match', 'value2')
SELECT таблица1.колонка1, таблица2.колонка1 FROM таблица1, таблица2 WHERE таблица1.колонка1 = таблица2.колонка1

### 3. Удаление с последующей проверкой
INSERT INTO таблица1 VALUES ('delete_me', 'data', 'info', 'desc')
DELETE FROM таблица1 WHERE таблица1.колонка1 = 'delete_me'
SELECT таблица1.колонка1 FROM таблица1 WHERE таблица1.колонка1 = 'delete_me'

## Примечания

- Все строковые значения должны быть в одинарных кавычках: 'значение'
- Названия таблиц и колонок чувствительны к регистру
- SQL ключевые слова (SELECT, FROM, WHERE, INSERT, DELETE) не чувствительны к регистру
- Первичный ключ добавляется автоматически и называется <table_name>_pk
- При вставке не нужно указывать значение первичного ключа - оно генерируется автоматически
- Количество значений в INSERT должно соответствовать количеству колонок (без учета первичного ключа)
